29class NodeManager(object):
30 """
31 Manages the Devices found on the SCOM bus.
32
33 Requirements:
34 - The NodeManager scans regularly for Devices on the SCOM bus of the Xcom-232i node and notifies
35 the NodeObservers about changes.
36 - The NodeManager is responsible to create an instance for every SCOM device found.
37 - The Node Manager needs to hold a list of present devices.
38 - In case a device appears on the SCOM bus it notifies the concerning observers with on_device_connected.
39 - In case a device disappears, it notifies the observers with onDeviceDisconnect.
40 - Scom RX errors are checked regularly and after too many errors the application gets terminated.
41 """

45 def __init__(self, node_name, scom=None, config=None, address_scan_info=None,

68 parity=config['scom']['parity'] if 'parity' in config['scom'] else serial.PARITY_NONE,

71 self._scom = studer_com

73 if address_scan_info:
74 self._address_scan_info = address_scan_info
75 else:
76 assert config, 'In case \'address_scan_info\' is not set the parameter config must be given!'
77 assert config.get('scom-device-address-scan'), 'Missing section \'scom-device-address-scan\' in config'

111 def subscribe(self, node_subscriber, filter_policy=('all',)):
112 """
113 Adds the given subscriber to the list.
114
115 :param node_subscriber Subscriber to be added to the list
116 :type node_subscriber DeviceSubscriber
117 :param filter_policy The list of device types of interest
118 :type filter_policy tuple[string]
119 :return True if subscriber could be added, otherwise False
120 """
121 # Subscribe the info about the new subscriber
122 self._subscribers.append({'subscriber': node_subscriber, 'filterPolicy': filter_policy})
123
124 # Notify already present devices to new subscriber
125 self._notify_subscriber(node_subscriber, filter_policy=filter_policy)
126 return True
127
128 def unsubscribe(self, node_subscriber):
129 """
130 Removes the given subscriber from the list.
131
132 :param node_subscriber Subscriber to be removed from the list
133 :type node_subscriber DeviceSubscriber
134 :return True if subscriber could be removed, otherwise False
135 """
136
137 for index, subscriber in enumerate(self._subscribers):
138 if subscriber['subscriber'] == node_subscriber:
139 self._subscribers.pop(index)
140 return True
141 return False

143 def _notify_subscriber(self, node_subscriber, filter_policy=('all',)):
144 """
145 Used to notify new subscribers about already present devices.
146 """
147 subscriber_info = {'subscriber': node_subscriber, 'filter_policy': filter_policy}
148
149 for deviceAddress, the_device in self._devices.items():
150 device_type = the_device.device_type
151 if device_type in subscriber_info['filter_policy'] or 'all' in subscriber_info['filter_policy']:
152 # Notify subscriber
153 subscriber_info['subscriber'].on_device_connected(the_device)
154
155 def _notify_subscribers(self, device, connected=True):
156 """
157 Notifies connect/disconnect of a device to all subscribers with the according filter policy.
158 """
159 # Notify subscribers about the device found
160 for subscriberInfo in self._subscribers:
161 device_type = device.device_type
162 # Apply subscribers filter policy
163 if device_type in subscriberInfo['filterPolicy'] or 'all' in subscriberInfo['filterPolicy']:
164 # Notify subscriber
165 if connected:
166 subscriberInfo['subscriber'].on_device_connected(device)
167 else:
168 subscriberInfo['subscriber'].on_device_disconnected(device)

195 def _run_with_exception_logging(self):
196 """Same as _run but logs exceptions to the console or log file.
197
198 This is necessary when running in testing/production environment.
199 In case of an exception thrown, the stack trace can be seen in the
200 log file. Otherwise there is no info why the thread did stop.
201 """
202 try:
203 self._run()
204 except Exception as e:
205 logging.error(e, exc_info=True)
206 finally:
207 # Wait here for a while. If leaving the method directly, the thread
208 # gets deleted and the is_alive() method won't work any more!
209 time.sleep(5)
210 return

212 def _run(self):
213 self.log.info(type(self).__name__ + ' thread running...')

215 while self._thread_should_run:

233 def _thread_sleep_interval(self, sleep_interval_in_seconds, decr_value=0.2):
234 """Tells the executing thread how long to sleep while being still reactive on _threadShouldRun attribute.
235 """
236 wait_time = sleep_interval_in_seconds
237
238 while wait_time > 0:
239 time.sleep(decr_value)
240 wait_time -= decr_value
241 # Check if thread should leave run loop
242 if not self._thread_should_run:
243 break

245 def _search_devices(self):
246 """Searches on the SCOM bus for devices.
247 """
248 assert len(self._address_scan_info), 'No device categories to scan found!'
249 need_garbage_collect = False
250
251 for device_type, addressScanRange in self._address_scan_info.items():
252 device_list = self._search_device_type(device_type, addressScanRange)
253
254 nbr_of_devices_found = len(device_list) if device_list else 0
255
256 if device_list:
257 for device_address in device_list:
258 # Check if device is present in device dict
259 if device_address in self._devices:
260 pass
261 else:
262 self._add_new_device(device_type, device_address)
263
264 # Compare number of instantiated devices (per category/group) and remove disappeared devices from list
265 if nbr_of_devices_found < self.get_number_of_instances(device_type):
266 self.log.warning(u'Some ScomDevices seem to be disappeared!')
267 missing_device_address_list = self._get_missing_device_addresses(device_type, device_list)
268
269 for missingDeviceAddress in missing_device_address_list:
270 missing_device = self._devices[
271 missingDeviceAddress] if missingDeviceAddress in self._devices else None
272 assert missing_device
273
274 self.log.info('Studer device disappeared: %s #%d' % (device_type, missingDeviceAddress))
275
276 # Notify subscribers about the disappeared device
277 self._notify_subscribers(device=missing_device, connected=False)
278
279 # Remove studer device from list
280 self._devices.pop(missingDeviceAddress)
281 need_garbage_collect = True
282
283 if need_garbage_collect: # Garbage collect to update WeakValueDictionaries
284 gc.collect()

286 def _add_new_device(self, device_type, device_address):
287 """Adds a new ScomDevice an notifies subscribers.
288 """
289 # Let the factory create a new SCOM device representation
290 self._devices[device_address] = DeviceFactory.create(device_type, device_address, self._scom)
291 self.log.info('Found new studer device: %s #%d' % (device_type, device_address))
292
293 # Notify subscribers about the device found
294 self._notify_subscribers(device=self._devices[device_address])

296 def _search_device_type(self, device_type, address_scan_range) -> [int]:
297 """Searches for devices of a specific category on the SCOM interface.
298
299 :return A list of device address found.
300 """
301 device_list = []
302 device_start_address = int(address_scan_range[0])
303 device_stop_address = int(address_scan_range[1])
304
305 self.log.info('Searching devices in group \'%s\'...' % device_type)
306
307 request_frame = BaseFrame(self.DEFAULT_RX_BUFFER_SIZE)
308
309 device_index = device_start_address
310 while device_index <= device_stop_address:
311 request_frame.initialize(src_addr=1, dest_addr=device_index)
312
313 prop = Property(request_frame)
314 # TODO For some devices 'parameter' value must be read instead of 'user info' (ex. RCC device)
315 prop.set_object_read(defines.OBJECT_TYPE_READ_USER_INFO,
316 DeviceType.get_reference_search_object_id(device_type),
317 defines.PROPERTY_ID_READ)
318
319 if request_frame.is_valid():
320 response_frame = self._scom.write_frame(request_frame, 0.5) # Set a short timeout during search
321
322 # if response_frame and response_frame.is_valid():
323 # For Testing Purpose
324 if True:
325 self.log.info('Found device on address: ' + str(device_index))
326 device_list.append(device_index)
327 else:
328 self.log.warning('Frame with error: ' + request_frame.last_error())
329
330 device_index += 1
331
332 if len(device_list) == 0:
333 self.log.warning('No devices in group \'%s\' found' % device_type)
334
335 return device_list

337 def _get_missing_device_addresses(self, device_type, device_address_list):
338 """"Searches in actually instantiated devices list (of a category) for devices not found in given device list.
339
340 :param device_type The device category in which to search for devices
341 :type device_type str
342 :param device_address_list List containing device address (which are still present and not missed)
343 :type device_address_list
344 """
345 missing_device_address_list = []
346
347 device_list = self.get_instances_of_category(device_type)
348
349 for internal_id, device in device_list.items():
350 if device.device_address not in device_address_list:
351 missing_device_address_list.append(device.device_address)
352
353 return missing_device_address_list

355 def _check_scom_rx_errors(self):
356 """Checks how many times there was an RX error on the SCOM bus.
357
358 After a few RX errors are detected, first a message is send (logged) and
359 after still more errors the application gets terminated.
360 """
361 msg = u'Scom bus no more responding!'
362 if self._scom.rxErrors > 50 and not self._scom_rx_error_message_send:
363 self.log.critical(msg)
364 self._scom_rx_error_message_send = True
365
366 if self._scom.rxErrors > 100:
367 sys.exit(msg)
