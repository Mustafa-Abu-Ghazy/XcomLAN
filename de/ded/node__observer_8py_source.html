65class NodeObserver(object):
66 """
67 Receives device notifications if NodeManager finds Studer devices.
68 (get notified about an SCOM device of interest.)
69 """
70
71 def __init__(self, node_manager):
72 super(NodeObserver, self).__init__()
73 assert node_manager, 'To create NodeObserver object, node_manager must be giving!'
74 self._node_manager = node_manager
75 self.log = logging.getLogger(__name__ + ":" + self.node_name)
76 self._connected_devices = {}
77 self._node_manager.subscribe(self)
78
79 def __del__(self):
80 self._node_manager.unsubscribe(self)
81

108 def on_device_connected(self, device):
109 """
110 Called whenever the DeviceManager detects a new SCOM device (of interest).
111
112 :param device:
113 :type device ScomDevice
114 :return: None
115 """
116 self._connected_devices[device.device_address] = device
117
118 def on_device_disconnected(self, device):
119 """
120 Called whenever the DeviceManager detects the disappearance of an SCOM device (of interest).
121
122 :param device:
123 :type device ScomDevice
124 :return: None
125 """
126 self._connected_devices.pop(device.device_address)
