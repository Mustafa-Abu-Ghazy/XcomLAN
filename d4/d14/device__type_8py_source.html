151 """

153 """
154 assert type(device_type) == cls, 'Unsupported Device Type'

156 if device_type == DeviceType.XTENDER:

158 elif device_type == DeviceType.COMPACT:
159 assert False, 'No Implementation of Device-Class for the Compact device type.'
160 elif device_type == DeviceType.VARIO_TRACK:
161 search_object_id = list(VARIO_TRACK_INFOS.keys())[0]
162 elif device_type == DeviceType.VARIO_STRING:
