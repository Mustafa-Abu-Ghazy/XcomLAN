58 """
59 enqueue method add the telemetry's information to the _telemetry_queue queue to be served in order by

79 def _send_nodes_telemetry_loop(self):
80 """
81 This method designed to be executed as a separate thread
82 """
83 log = logging.getLogger(__name__ + ":" + "send_nodes_telemetry_loop")
84 while True:
85 if self._telemetry_restored_queue.qsize():
86 telemetry = self._telemetry_restored_queue.get()
87 else:
88 telemetry = self._telemetry_queue.get()
89 try:
90 # Try to connect to the Server if not connected yet
91 if not self.is_connected():
92 self.connect()
93 # Try sending telemetry to the server
94 self.send_node_telemetry(
95 node_name=telemetry["node_name"],
96 timestamp=telemetry["timestamp"],
97 telemetry_values=telemetry["telemetry_values"]
98 )
99 except Exception as e:
100 self._telemetry_restored_queue.put(telemetry) # Restore the telemetry back to the restored_queue
101 log.exception(e) # log the exception
102 sleep(30) # Sleep 30 Seconds

149 @classmethod
150 def _generate_dict_header_for_csv_log_file(cls, csv_log_file_path):
151 # supports only nodes with single Xtender device
152 first_row, second_row, third_row = list(csv.reader(open(csv_log_file_path, 'r')))[0:3]
153 header = ['ts'] + [''] * (len(first_row) - 2)
154
155 for idx in range(1, len(first_row) - 1):
156 if first_row[idx] == '':
157 first_row[idx] = first_row[idx - 1]
158
159 if first_row[idx].startswith('XT') or first_row[idx].startswith('DEV XT'):
160 header[idx] = 'XT' + '1' + second_row[idx]
161 elif first_row[idx].startswith('VS') or first_row[idx].startswith('DEV VS'):
162 header[idx] = 'VS' + third_row[idx] + second_row[idx]
163 elif first_row[idx].startswith('VT') or first_row[idx].startswith('DEV VT'):
164 header[idx] = 'VT' + third_row[idx] + second_row[idx]
165 elif first_row[idx].startswith('BSP'):
166 header[idx] = 'BSP' + second_row[idx]
167 elif first_row[idx].startswith('DEV'):
168 header[idx] = 'DEV' + second_row[idx]
169 elif first_row[idx] == 'Solar power (ALL) [kW]':
170 header[idx] = 'SolarPowerALL' + second_row[idx]
171 else:
172 header[idx] = re.sub('[^0-9a-zA-Z]+', '', first_row[idx]) + third_row[idx] + second_row[idx]
173 return header

188 def push_csv_input_to_tb(self, node_name, csv_log_file_path, feed_as_realtime=False):
189 """
190 Read Telemetry from CSV file of node and Push them to TB Server
191
192 :param feed_as_realtime: if set the Telemetry will be parsed and send as a realtime
193 neglecting the stored timestamp
194 """
195 log = logging.getLogger(__name__ + ":" + self.node_profile + ":" + node_name)
196
197 # Connect The Device
198 # device_type: will be used only when creating not existing device
199 self.gw_connect_device(node_name, device_type=self.node_profile)
200
201 telemetry = {}
202 attributes = {}
203
204 # Define Fields Names
205 fieldnames = self._generate_dict_header_for_csv_log_file(csv_log_file_path)
206
207 # Creating A CSV DictReader object
208 csv_dict_reader = list(csv.DictReader(open(csv_log_file_path, 'r'), fieldnames=fieldnames))
209
210 # Read (CSVDictReader), Skip First 3 Rows. Focus on Telemetries
211 for index, row in enumerate(csv_dict_reader[3:1443]):
212 try:
213 logging.info('CSV ROW Number: ' + str(index + 4))
214 date_time_str = row['ts']
215 date_time_obj = datetime.strptime(date_time_str, '%d.%m.%Y %H:%M')
216 timestamp = date_time_obj.timestamp()
217 telemetry['ts'] = int(timestamp * 1000) if not feed_as_realtime else int(round(time() * 1000))
218 telemetry['values'] = {i: self._parse_value_from_csv_log_file(row[i]) for i in fieldnames[1:]}
219
220 result = self.gw_send_telemetry(node_name, telemetry)
221 result_status = result.get() == TBPublishInfo.TB_ERR_SUCCESS
222 log.info('Node: ' + node_name + ' Telemetry Content: ' + str(telemetry))
223 log.info('Node: ' + node_name + ' File: ' + csv_log_file_path +
224 ' Send Telemetry Result Status: ' + str(result_status))
225 except Exception as e:
226 logging.exception(e)
227
228 # Read (CSVDictReader), Skip First 1443 Rows. Focus on Attributes
229 for index, row in enumerate(csv_dict_reader[1443:]):
230 if re.match(r'[PI][0-9]+', row[fieldnames[0]]):
231 attributes[row[fieldnames[0]]] = self._parse_value_from_csv_log_file(row[fieldnames[1]])
232
233 result = self.gw_send_attributes(node_name, attributes)
234 result_status = result.get() == TBPublishInfo.TB_ERR_SUCCESS
235 log.info('Node: ' + node_name + ' File: ' + csv_log_file_path + ' Attributes Content: ' + str(attributes))
236 log.info('Node: ' + node_name + ' Send Attributes Result Status: ' + str(result_status))
237
238 # Disconnect The Device
239 self.gw_disconnect_device(node_name)
240
241 if feed_as_realtime:
242 sleep(60)
