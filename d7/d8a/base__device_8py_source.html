107 def software_version(self):
108 """
109 Returns the software version.
110
111 :return The Software version as dict {major, minor, patch}, zeros for the device types haven't related UserInfo
112 :rtype dict or None
113 """
114 id_soft_msb = id_soft_lsb = None
115 if hasattr(self, 'read_user_info_smsb'):
116 id_soft_msb = getattr(self, 'read_user_info_smsb')()
117 if hasattr(self, 'read_user_info_slsb'):
118 id_soft_lsb = getattr(self, 'read_user_info_slsb')()
119 if id_soft_msb and id_soft_lsb:
120 id_soft_major_version = int(id_soft_msb) >> 8
121 id_soft_minor_version = int(id_soft_lsb) >> 8
122 id_soft_revision = int(id_soft_lsb) & 0xFF
123 return {'major': id_soft_major_version, 'minor': id_soft_minor_version, 'patch': id_soft_revision}
124 return {'major': 0, 'minor': 0, 'patch': 0}

127 """
128 Reads a user info on the device.
129
130 :param parameter_id
131 :type parameter_id int
132 :return The parameter read
133 :type return bytearray
134 """
135 value = bytearray()
136 request_frame = ScomFrame()
137
138 request_frame.initialize(src_addr=1, dest_addr=self.device_addressdevice_address, data_length=99)
139
140 prop = Property(request_frame)
141 prop.set_object_read(OBJECT_TYPE_READ_USER_INFO, parameter_id, PROPERTY_ID_READ)
142
143 if request_frame.is_valid():
144 response_frame = self._get_scom().write_frame(request_frame) # Method call is blocking
145
146 if response_frame:
147 if response_frame.is_valid():
148 value_size = response_frame.response_value_size()
149 value = response_frame[24:24 + value_size]
150 elif response_frame.is_data_error_flag_set():
151 self.log.warning('Warning: Error flag set in response frame! will reading UserInfo %d' %
152 parameter_id)
153 else:
154 self.log.warning('No response frame received!')
155 else:
156 self.log.warning('Request frame not valid')
157
158 return value

160 def read_user_info(self, user_info_id):
161 """
162 Uses the userInfoTable to access the needed user info. and get it's value.
163
164 :return The value received from the device
165 :rtype: float, (enum), None.
166 """
167 user_info = self._userInfosTable.get(user_info_id)
168
169 if not user_info:
170 self.log.warning("Could not read UserInfo %d It's UnSupported UserInfo" % (user_info_id,))
171 return None
172
173 # For Testing Purpose
174 # return 0
175
176 value = self._read_user_info_by_parameter_id(user_info_id)
177 if value:
178 if user_info["Format"] == PropertyFormat.FLOAT:
179 assert len(value) == 4, 'Length of Value: Expected 4 got %d' % len(value)
180 value = struct.unpack('f', value[0:4])[0]
181 elif user_info["Format"] == PropertyFormat.SHORT_ENUM:
182 assert len(value) == 2, 'Length of Value: Expected 2 got %d' % len(value)
183 value = struct.unpack('H', value[0:2])[0] # Read 'ENUM' format as unsigned int (H)
184 else:
185 assert False, 'Unsupported format for this UserInfo'
186 return value
187 else:
188 self.log.warning("Could not read UserInfo %d '%s'" % (user_info_id, user_info["Short desc."]))
189 return None

312 def _generate_short_desc_string_as_function_name(cls, short_desc_string):
313 """
314 Converters the UserInfo 'Short desc.' string to a string compatible to be used as part of function name
315
316 :param short_desc_string:
317 :return the new_short_desc_string
318 :rtype: str
319 """
320 new_short_desc_string = None
321 if short_desc_string.strip():
322 new_short_desc_string = short_desc_string.strip()
323 new_short_desc_string = new_short_desc_string.replace("+", "_plus")
324 new_short_desc_string = new_short_desc_string.replace("-", "_minus")
325 new_short_desc_string = re.sub("[^0-9a-zA-Z]+", "_", new_short_desc_string)
326 new_short_desc_string = re.sub("[_]*$", "", new_short_desc_string)
327 return new_short_desc_string.lower()

349 def set_user_infos_table(self, user_infos_table):

351 Sets the _userInfosTable to the passed dict and dynamically generates getters functions for all the infos

355 - read_user_info_i3001(), read_user_info_tbat() --> for the Xtender Info 3001

359 :return:
360 """
361
362 self._userInfosTable = user_infos_table
363
364 _list_of_all_short_desc_strings = [_user_info_dict["Short desc."]
365 for _, _user_info_dict

369 exec("setattr(self, 'read_user_info_i{0}', self._create_read_user_info_function({0}))".format(user_info_id))
370 if _list_of_all_short_desc_strings.count(user_info_dict["Short desc."]) == 1:
371 short_desc_string = self._generate_short_desc_string_as_function_name(user_info_dict["Short desc."])
