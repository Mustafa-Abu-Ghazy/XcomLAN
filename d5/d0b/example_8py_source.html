152if __name__ == "__main__":
153 things_board_client = ThingsBoardClient(THINGSBOARD_SERVER, GATEWAY_Access_TOKEN)
154
155 nodes_observers = []
156 for node_name, node_dict in NODES_LIST.items():
157 try:
158 # Try to create NodeManager & NodeObserver and connect to the interface
159 if node_dict["interface"]:
160 # Prepare SCOM Node Configuration
161 config = XCOM_LAN_CONFIG_TEMPLATE
162 config["scom"]["interface"] = node_dict.get("interface")
163 # Create node manager detecting devices on the SCOM bus
164 node_manager = NodeManager(node_name=node_name, config=config)
165 # Create observer waiting for NodeManager notifications
166 node_observer = NodeObserver(node_manager)
167 nodes_observers.append(node_observer)
168 except Exception as e:
169 logging.error("Can't connect to the node: " + node_name)
170 logging.exception(e)
171
172 # This loop reads the required infos from (XT, VT, BSP) and push it to ThingsBoard
173 while True:
174 for node_observer in nodes_observers:
175 # Use Copy to avoid "RuntimeError: dictionary changed size during iteration"
176 connected_devices = copy.copy(node_observer.connected_devices)
177 for device_address, device in connected_devices.items():
178 thread = threading.Thread(target=invoke_read_user_infos_as_telemetry, args=(
179 things_board_client,
180 node_observer.node_name,
181 device,
182 get_list_of_user_infos_of_interest(device)
183 ))
184 thread.setDaemon(True)
185 thread.start()
186
187 time.sleep(READINGS_DELAY_IN_SECONDS)
